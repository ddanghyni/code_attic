from torch import nn

class SIMPLE_CNN(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Sequential(
            nn.Conv2d(1, 3, kernel_size=5),   # (28,28,1) -> (24,24,3)
            nn.SiLU(),                         # swish = SiLU
            nn.MaxPool2d(2),                   # -> (12,12,3)
            nn.Conv2d(3, 6, kernel_size=5),    # -> (8,8,6)
            nn.SiLU(),
            nn.MaxPool2d(2),                   # -> (4,4,6)
        )
        self.fc1 = nn.Sequential(
            nn.Flatten(),                      # -> 4*4*6 = 96, FC layer을 통과하기 위함.
            nn.Linear(4 * 4 * 6, 84),
            nn.SiLU(),
            nn.Linear(84, 10),
        )

    def forward(self, x):
        x = self.conv1(x)
        x = self.fc1(x)
        return x

from torch import nn

class LeNet_5(nn.Module):
    def __init__(self):
        super().__init__()
        
        self.conv1 = nn.Sequential(nn.Conv2d(1, 6, 5, stride=1, padding=2), # 논문에선 32x32짜리의 데이터를 예시로 했음
                                   nn.SiLU()) #swish                        # 우리가 사용하는 MNIST는 28x28이므로 zero-padding 2 추가!
        
        self.Maxpool1 = nn.MaxPool2d(2, 2)
        
        self.conv2 = nn.Sequential(nn.Conv2d(6, 16, 5),
                                   nn.SiLU())
        
        self.Maxpool2 = nn.MaxPool2d(2, 2)
        
        self.fc = nn.Sequential(nn.Linear(16 * 5 * 5, 120),
                                nn.SiLU(),
                                nn.Linear(120, 84),
                                nn.SiLU(),
                                nn.Linear(84, 10))
        
    def forward(self, x):
        x = self.conv1(x)
        x = self.Maxpool1(x)
        x = self.conv2(x)
        x = self.Maxpool2(x)
        x = torch.flatten(x, start_dim=1)
        x = self.fc(x)
        return x

LeNet5를 구현하기 위해 한 가지 중요한 점이 있다.

우선 논문에서 아키텍처를 구현할때 사용한 데이터는 32x32 데이터이다.

하지만 우리가 쓰는 MNIST 데이터는 아래에서 언급하겠지만 28x28 데이터이다.

따라서 논문과 같은 셋팅을 위해선 첫 Conv layer에 zero-padding 2를 추가하여 shape을 맞춰줄 필요가 있다.

위의 코드를 보면

self.conv1 = nn.Sequential(nn.Conv2d(1, 6, 5, stride = 1, padding = 2))

를 보면 알 수 있다.

여기서 zero-padding은 how to pad the input volume with zeros around the border 이다.

input data의 경계에 0의 값은 가진 값을 둘러주는 것이다.

이런식으로 zero-padding은 사이즈를 맞춰주는 역할도 있지만

다음과 같은 상황을 방지하기 위해서 많이 사용된다.

Consider a 32X32 input convolved repeadly with 5X5 filters... then volume will shrinks spatially (32->28->24...)

Shrinking too fast -> not good --> 이게 왜 안좋을걸까? --> input(output) shrink fast --> layer 수 적어임 --> 깊은 network 생성 불가능 --> 일반화 bad

def Train(model, train_DL, val_DL, criteron, optimizer, EPOCH):
    loss_history = []
    acc_history = []
    val_loss_history = []
    val_acc_history = []
    NoT = len(train_DL.dataset)
    NoV = len(val_DL.dataset)

    for ep in range(EPOCH):
        # model을 훈련모드로 전환 --> dropout, BatchNorm 등 훈련과 test에서 작용 방식이 다른 놈들을 위한
        model.train()
        rloss = 0
        rcorrect = 0
        for x_batch, y_batch in train_DL:
            x_batch = x_batch.to(DEVICE)
            y_batch = y_batch.to(DEVICE)
            # inference
            y_hat = model(x_batch)
            # loss
            loss = criteron(y_hat, y_batch)
            # update
            optimizer.zero_grad() # 각 epoch마다 gradient 누적방지
            # backprop
            loss.backward()
            # weight update
            optimizer.step()
            # loss accum
            loss_b = loss.item() * x_batch.shape[0]
            rloss += loss_b
            # accuracy accum
            pred = y_hat.argmax(dim = 1)
            correct_b = torch.sum(pred == y_batch).item()
            rcorrect += correct_b

        loss_e = rloss / NoT
        loss_history += [loss_e]
        accuracy_e = rcorrect / NoT * 100
        acc_history += [accuracy_e]

        # Vali
        model.eval()
        val_rloss = 0
        val_rcorrect = 0
        with torch.no_grad():
            for x_batch, y_batch in val_DL:
                x_batch = x_batch.to(DEVICE)
                y_batch = y_batch.to(DEVICE)
                # inference
                y_hat = model(x_batch)
                # loss
                val_loss = criteron(y_hat, y_batch)
                # loss accum
                val_loss_b = val_loss.item() * x_batch.shape[0]
                val_rloss += val_loss_b
                # accuracy accum
                pred = y_hat.argmax(dim = 1)
                correct_b = torch.sum(pred == y_batch).item()
                val_rcorrect += correct_b

        val_loss_e = val_rloss / NoV
        val_loss_history += [val_loss_e]
        val_accuracy_e = val_rcorrect / NoV * 100
        val_acc_history += [val_accuracy_e]

        print(f"Epoch: {ep+1}, train loss: {round(loss_e, 3)}, val loss: {round(val_loss_e, 3)}")
        print(f"Train Accuracy: {rcorrect}/{NoT} ({round(accuracy_e, 1)}%)")
        print(f"Val Accuracy: {val_rcorrect}/{NoV} ({round(val_accuracy_e, 1)}%)")
        print("-"*20)

    return loss_history, acc_history, val_loss_history, val_acc_history

y_hat.argmax(dim = 1)¶

예를 들어서 사진 X을 모델에 통과해서 4종류의 동물을 구분하는 분류 문제를 푼다고 하자.

데이터가 모델을 통과 -> logit -> softmax(다중분류) -> 확률이 나올것이다.

y_hat = 
[[0.1, 0.7, 0.05, 0.15],   # 샘플 0
 [0.6, 0.1, 0.2,  0.1 ],   # 샘플 1
 [0.2, 0.1, 0.6,  0.1 ]]   # 샘플 2

위와 같이 확률 값이 output으로 나올거다.

우선 샘플0만을 보면

  개   고양이  사슴   말
[[0.1, 0.7, 0.05, 0.15]]

이라고 생각하면 쉽다.

그럼 샘플0 즉 첫번째 사진 X_0는 확률이 제일 높은 0.7 고양이 사진이라고 모델이 판단하게 되는것이다.

단순히 y_hat의 정보를 위의 matrix로 보는것보다.

argmax(dim=1)을 사용해서 각 행에서 가장 큰 값의 인덱스를 반환하게 하자.

pred = y_hat.argmax(dim=1)

pred = [1, 0, 2] ==> 이런식으로 각 샘플에서 가장 큰 값의 인덱스를 받을 수 있다.

이제 이를

torch.sum(pred == y_batch).item()

y_batch와 비교하여 sum을 하면 모델이 맞춘 정답의 수를 알 수 있다.

def Test(model, test_DL):
    #test 모드로 전환
    model.eval()
    
    with torch.no_grad():
        rcorrect = 0
        for x_batch, y_batch in test_DL:
            x_batch = x_batch.to(DEVICE)
            y_batch = y_batch.to(DEVICE)
            
            #inference
            y_hat = model(x_batch)
            pred = y_hat.argmax(dim = 1)
            correct_b = torch.sum(pred == y_batch).item()
            rcorrect += correct_b
        accuracy_e = rcorrect / len(test_DL.dataset) * 100
    print(f"Test Accuaracy: {rcorrect}/{len(test_DL.dataset)} ({round(accuracy_e, 1)}%)")

def Test_plot(model, test_DL):
    model.eval()
    with torch.no_grad():
        x_batch, y_batch = next(iter(test_DL))
        x_batch = x_batch.to(DEVICE)
        y_hat = model(x_batch)
        pred = y_hat.argmax(dim=1)

    x_batch = x_batch.to("cpu")

    plt.figure(figsize=(8,4))
    for idx in range(6):
        plt.subplot(2,3, idx+1, xticks=[], yticks=[])
        plt.imshow(x_batch[idx].permute(1,2,0).squeeze(), cmap="gray")
        pred_class = test_DL.dataset.classes[pred[idx]]
        true_class = test_DL.dataset.classes[y_batch[idx]]
        plt.title(f"{pred_class} ({true_class})", color = "g" if pred_class==true_class else "r")

먼서 overfitting을 방지하는 법을 설명하기 전에 ML에서 우리가 싸워야할 가장 큰 두 적에 해서 말해보겠다.

  1. underfitting

underfitting은 일반적으로 bias가 main error이다. 이는 모델을 훈련할때 다른 optimization 방법을 쓰면서 해결할 수 있다.

대표적으론 EWMA와 모멘텀을 이용한 Adam이 있다.

  1. overfitting

overfitting은 일반적으로 variance가 main error이다. 이는 모델을 훈련할때 다른 regularzation을 이용하면 된다.

규제의 방법으론 model에 constraints를 추가, objective function에 extra terms을 추가 그리고 ensemble methdos를 이용하는 것이 있다.

DL에선 대표적으로 Early stopping, 일종의 bagged ensemble methods인 Dropout 등이 있다.

# SIMPLE MLP MODEL

from torch import nn

class MLP(nn.Module):
    def __init__(self):
        super().__init__()
        
        self.linear = nn.Sequential(nn.Linear(28 * 28, 80),
                                    nn.ReLU(),
                                    nn.Linear(80, 10))
        
    def forward(self, x):
        x = torch.flatten(x, start_dim=1) 
        x = self.linear(x)
        return x
